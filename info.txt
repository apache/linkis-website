创建新版本
基于当前的1.1.2 创建1.1.3

#存档1.1.2 版本文档
npm run docusaurus docs:version-1.1.2

#拷贝中文新版本文档
cd i18n/zh-CN/docusaurus-plugin-content-docs
cp -r current  version-1.1.2
cp -r current.json version-1.1.2.json

修改配置docusaurus.config.js

 versions: {
        current: {
          path: '1.1.2',
          label: 'Next(1.1.2)'
        },
        '1.1.1': {
          path: 'latest',
        },
      }
